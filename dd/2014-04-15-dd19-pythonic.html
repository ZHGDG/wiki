题面儿:

  1. 从1到100
  2. 如果能被3和5同时整除,吼:FizzBuzz
  3. 如果只能被3整除,吼:Fizz
  4. 如果只能被5同时整除,吼:Buzz

Javacious

{% highlight python %} import sys

class Value(object): def init(self,value): self.setValue(value)

def setValue(self,value):
    self.value = value

def getValue(self):
    return self.value

def toString(self):
    return self.getValue().__str__()

class FizzBuzz(object): def init(self, n): if n % 15 == 0: value = 'FizzBuzz'; elif n % 3 == 0: value = 'Fizz'; elif n % 5 == 0: value = 'Buzz'; else: value = str(n); self.setValue(value);

def setValue(self,value):
    self.value = Value(value);

def getValue(self):
    return self.value;

class FizzBuzzRunner(object): def init(self, n): self.setN(n)

def setN(self, n):
    self.n = n

def run(self):
    for i in range(1,self.n):
        sys.stdout.write(FizzBuzz(i).getValue().toString()+'\n');

if name == 'main': n = 101; FizzBuzzRunner(n).run()
